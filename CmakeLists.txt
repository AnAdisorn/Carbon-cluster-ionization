cmake_minimum_required(VERSION 3.10)

project(carbon-cluster-ionization VERSION 0.0.1)

############################################################
# Compiler
############################################################

set(CMAKE_CXX_STANDARD 20)

add_compile_options(-Wall -Wextra)
set(CMAKE_CXX_FLAGS "-O3")

############################################################
# Create a library
############################################################

add_library(particle_container STATIC src/particle_container.cpp)
add_library(particle_interactions STATIC src/particle_interactions.cpp)
add_library(field_interactions STATIC src/field_interactions.cpp)
add_library(parameters STATIC src/parameters.cpp)

target_include_directories(particle_container
    PUBLIC 
        include
)
target_include_directories(particle_interactions
    PUBLIC 
       include
)
target_include_directories(field_interactions
    PUBLIC 
       include
)
target_include_directories(parameters
    PUBLIC 
       include
)

############################################################
# Create an executable
############################################################
